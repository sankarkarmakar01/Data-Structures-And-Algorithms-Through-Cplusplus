class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        vector<int> ans(2);
        int first = binarySearchForFirstAndLastPosition(nums, target, true);
        int last = binarySearchForFirstAndLastPosition(nums, target, false);
        ans[0] = first;
        ans[1] = last;
        return ans;
    }

    int binarySearchForFirstAndLastPosition(vector<int> &arr, int target, bool isFirst)
    {
        int ans = -1;
        int start = 0;
        int end = arr.size() - 1;
        int mid;
        while(start <= end)
        {
            mid = start + (end - start)/2;
            if (arr[mid] == target)
            {
                ans = mid;
                if(isFirst)
                {
                    end = mid - 1;
                }
                else 
                {
                    start = mid + 1;
                }
            }
            else if(arr[mid] < target)
            {
                start = mid + 1;
            }
            else
            {
                end = mid - 1;
            }
        }
        return ans;
    }
};
